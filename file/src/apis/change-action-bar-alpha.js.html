    import {
// ! windvane无此className
} from '../constants';
import { generateAPI } from '../utils/generate-api';

/**
 *  更改 App 顶部 ActionBar 文字透明度   注：调用该方法之后文字会变成白色！
 * @param {number} params.alpha  - 0-100
 * @returns {Promise<*>}
 *
 * @example
 *  lib.exec('change-action-bar-alpha', {
 *  params: { alpha: 30 }
  }).then((info) => {
    alert(`[change-action-bar-alpha][then] ${JSON.stringify(info, null, 2)}`);
  }).catch((err) => {
    alert(`[change-action-bar-alpha][error] ${JSON.stringify(err, null, 2)}`);
  });
 */
export const changeActionBarAlpha = generateAPI({
  wv: {
    ios: {
      className: '',
      methodName: 'refreshActionBarAlpha',
    },
  },
  ios: {
    bridgeName: 'refreshActionBarAlpha',
  },
  android: {
    bridgeName: 'refreshActionBarAlpha',
  },
});
