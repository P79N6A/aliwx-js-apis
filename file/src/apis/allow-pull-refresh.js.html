    src/apis/allow-pull-refresh.js

    import {
  WV_CLS_IOS_CUSTOM,
} from '../constants';
import { generateAPI } from '../utils/generate-api';

async function preParam(params = {}) {
  const { status } = params;
  return { status: status ? 1 : 0 };
}

/**
 * 是否允许书城下拉刷新
 *
 * 支持平台：书旗Android、书旗iOS
 *
 * @param {number} params.status - 是否允许下拉刷新 0/1
 * @returns {Promise<*>}
 *
 * @example
 *  libSQ.exec('allow-pull-refresh', {
 *     params: {
 *      status: 1
 *     }
 *  }).then((info) => {
 *    alert(`[refresh][allow-pull-refresh][then] ${JSON.stringify(info, null, 2)}`);
 *  }).catch((err) => {
 *    alert(`[refresh][allow-pull-refresh][error] ${JSON.stringify(err, null, 2)}`);
 *  });
 *
 */
export const allowPullRefresh = generateAPI({
  wv: {
    ios: {
      className: WV_CLS_IOS_CUSTOM,
      methodName: 'allowPullFresh',
      preParam,
    },
  },
  ios: {
    bridgeName: 'allowPullFresh',
    preParam,
  },
  android: {
    bridgeName: 'allowPullFresh',
    preParam,
  },
});
