    /**
 * 单个环境下的 API 配置
 *
 * @typedef {Object} APIConfig
 * @property {function} preParam - 对传入参数的处理函数
 * @property {function} exec - API 底层执行函数，WindVane 默认使用{@link execWV}，普通 JS 桥默认使用 {@link execLegacy}
 * @property {function} postResult - 对最终结果的处理函数
 * @property {string} [className] - WindVane 调用时需要的 className
 * @property {string} [methodName] - WindVane 调用时需要的 methodName
 * @property {string} [bridgeName] - 普通调用时需要的 JS 桥的名字
 */

/**
 * 多个环境下的 API 配置合集
 *
 * @typedef {Object} APIConfigs
 * @property {Object} wv - WindVane 下的 API 配置合集
 * @property {APIConfig} wv.ios - WindVane 下 iOS 的 API 配置
 * @property {APIConfig} wv.android - WindVane 下 Android 的 API 配置
 * @property {APIConfig} wv.ios_tmReader - 天猫读书 WindVane 下 iOS 的 API 配置
 * @property {APIConfig} wv.android_tmReader - 天猫读书 WindVane 下 Android 的 API 配置
 * @property {APIConfig} android - Android 的 API 配置
 * @property {APIConfig} ios - iOS 的 API 配置
 * @property {Object} miniProgram - 小程序 下的 API 配置合集
 * @property {APIConfig} miniProgram.tb_tmReader - 小程序 下的 API 配置合集
 * @property {APIConfig} miniProgram.aliPay_sqReader - 小程序 下的 API 配置合集
 */

import {
  INTERNAL_API_SUPPORTS,
  INTERNAL_RAW_ERROR,
  SUPPORT_ANDROID,
  SUPPORT_IOS,
  SUPPORT_WINDVANE_ANDROID,
  SUPPORT_WINDVANE_IOS,
  WV_SUCCESS_CODES,
  SUPPORT_WINDVANE_IOS_TMREADER,
  SUPPORT_WINDVANE_ANDROID_TMREADER,
  SUPPORT_MINIPROGRAM_TB_TMREADER,
  SUPPORT_MINIPROGRAM_ALIPAY_SQREADER,
  MINI_ALIPAY_EVENT,
} from '../constants';
import { getGlobalEnv } from './global';
import { unifiedErrorHandler } from './unified-error-hander';
import { execLegacy, execWV, execMiNi } from './exec';
import { isUndefined, isFunction } from './is-a';
import { generateSDKError } from './generate-sdk-error';
import { dispatchAsEvent } from './events';
import { warn } from './log';

/**
 * 生成 API
 *
 * @param {APIConfigs} apiConfigs - 多环境下的 API 配置合集，根据传入的 key，自动添加 supports
 * @returns {Promise<void>}
 */
export async function generateAPI(apiConfigs) {
  const { wv = {}, android, ios, miniProgram = {} } = apiConfigs;
  const wvIOS = wv.ios; // iOS
  const wvAndroid = wv.android; // android
  const wvIOS_tmReader = wv.ios_tmReader;// 天猫读书iOS
  const wvAndroid_tmReader = wv.android_tmReader;// 天猫读书android
  const miniTMReader = miniProgram.tb_tmReader; // 天猫小程序
  const miniSQReader = miniProgram.aliPay_sqReader;// AliPay 小程序
  const supports = [];

  if (android) {
    supports.push(SUPPORT_ANDROID);
  }
  if (ios) {
    supports.push(SUPPORT_IOS);
  }
  if (wvIOS) {
    supports.push(SUPPORT_WINDVANE_IOS);
  }
  if (wvAndroid) {
    supports.push(SUPPORT_WINDVANE_ANDROID);
  }
  if (wvIOS_tmReader) {
    supports.push(SUPPORT_WINDVANE_IOS_TMREADER);
  }
  if (wvAndroid_tmReader) {
    supports.push(SUPPORT_WINDVANE_ANDROID_TMREADER);
  }
  if (miniTMReader) {
    supports.push(SUPPORT_MINIPROGRAM_TB_TMREADER);
  }
  if (miniSQReader) {
    supports.push(SUPPORT_MINIPROGRAM_ALIPAY_SQREADER);
  }

  const apiFn = async function _apiFn(params = {}) {
    const env = await getGlobalEnv();

    if (env.isSQReader) { // 书旗客户端相关
      if (env.isIOS) {
        if (env.isInWV && wvIOS) {
          return wvWrapper(env, wvIOS, params);
        } else if (ios) {
          return legacyWrapper(env, ios, params);
        }
      } else if (env.isAndroid) {
        if (env.isInWV && wvAndroid) {
          return wvWrapper(env, wvAndroid, params);
        } else if (android) {
          return legacyWrapper(env, android, params);
        }
      }
    } else if (env.isTMReader) { // 天猫客户端相关，天猫读书都是windwane
      if (env.isIOS && wvIOS_tmReader) {
        return wvWrapper(env, wvIOS_tmReader, params);
      } else if (env.isAndroid && wvAndroid_tmReader) {
        // tm android windwane的方式和sq android非windwane调用方式相同，都是对象找方法
        return legacyWrapper(env, wvAndroid_tmReader, params);
      }
    } else if (env.isMiNi_TB_tmReader) { // 天猫小程序相关
      return miniWrapper(env, miniTMReader, params);
    } else if (env.isMiNi_AliPay_sqReader) { // 书旗小程序相关
      return miniWrapper(env, miniSQReader, params);
    }

    throw generateSDKError('ERR_API_UNAVAILABLE');
  };
  apiFn[INTERNAL_API_SUPPORTS] = supports;
  return apiFn;
}

/**
 * 真正的桥的调用
 * @param env
 * @param config
 * @param params
 * @returns {Promise.<*>}
 */
async function genericLifeCycle(env, config, params = {}) {
  let { preParam, exec, postResult } = config;

  if (!isFunction(preParam)) {
    preParam = async p => [p];
  }
  if (!isFunction(postResult)) {
    postResult = async r => r;
  }
  if (!isFunction(exec)) {
    if (env.isMiNi_TB_tmReader || env.isMiNi_AliPay_sqReader) {
      exec = execMiNi;
    } else if (env.isInWV) {
      exec = execWV;
    } else if (env.isAndroid || env.isIOS) {
      exec = execLegacy;
    }
  }

  // 准备入参,调用桥
  const preProcessedParam = await preParam(params);
  const result = await unifiedErrorHandler(exec, ...preProcessedParam);
  return postResult(result, params);
}

/**
 * ios windwane调用的桥
 * @param env
 * @param config
 * @param params
 * @returns {Promise}
 */
async function wvWrapper(env, config, params = {}) {
  let { preParam, postResult } = config;
  const { exec, className, methodName } = config;

  // TODO validation
  // className, methodName

  if (!isFunction(preParam)) {
    preParam = async p => p;
  }

  if (!isFunction(postResult)) {
    postResult = async r => r;
  }

  return genericLifeCycle(env, {
    preParam: async (p) => {
      const preProcessedParam = await preParam(p);
      return [{
        className,
        methodName,
        params: preProcessedParam,
        timeout: params.timeout,
      }];
    },
    exec,
    postResult: async (r, p) => {
      if (r && WV_SUCCESS_CODES.indexOf(r.ret) !== -1) {
        if (isUndefined(r.data)) {
          return postResult(r, p);
        } else {
          return postResult(r.data, p);
        }
      }
      const err = generateSDKError('ERR_API_UNKNOWN');
      err.detail = {
        [INTERNAL_RAW_ERROR]: r,
      };

      throw err;
    },
  },
  params);
}

/**
 * 端上的桥调用
 * @param env
 * @param config
 * @param params
 * @returns {Promise}
 */
async function legacyWrapper(env, config, params = {}) {
  let { preParam, postResult, exec } = config;
  const { bridgeName } = config;

  //  android windwane调用方式和ios 不同，提前设置function，以免genericLifeCycle中设置错误
  if (!isFunction(exec)) {
    exec = execLegacy;
  }

  // TODO validation
  // bridgeName
  if (!isFunction(preParam)) {
    preParam = async p => p;
  }

  if (!isFunction(postResult)) {
    postResult = async r => r;
  }

  return genericLifeCycle(env, {
    preParam: async (p) => {
      console.log('genericLifeCycle a');
      const preProcessedParam = await preParam(p);
      // 特殊处理 exec 桥
      if (bridgeName === 'exec') {
        return [bridgeName, ...preProcessedParam];
      }

      if (config.noParam) {
        return [bridgeName];
      }

      return [bridgeName, preProcessedParam];
    },
    exec,
    postResult: async (r, p) => {
      let parsedResult;
      console.log('genericLifeCycle b');

      try {
        parsedResult = JSON.parse(r);
      } catch (e) {
        warn(e);
        parsedResult = r;
      }

      return postResult(parsedResult, p);
    },
  },
  params);
}
