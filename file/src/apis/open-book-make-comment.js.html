    /**
 * Create by 经笥 on 2018/9/9
 */

import {
  WV_CLS_IOS_NAVIGATOR,
} from '../constants';
import { encodeParams } from '../utils/exec';
import { generateAPI } from '../utils/generate-api';

async function preParam(params = {}) {
  const { bookId } = params;

  return encodeParams({ commentListBookId: bookId });
}

/**
 *  打开客户端 写书评页面
 *
 * @param {Object} params - 调用客户端桥所需要的参数
 * @param {string} params.bookId - 书籍Id
 *
 * @returns {Promise<*>}
 *
 * @example 调用示例
 * document.addEventListener('lib-sq-ready', (evt) => {
 *    const {lib} = evt.detail;
 *    lib.exec('open-book-make-comment',{ params: {bookId: '6479275'} }).then(function (info) {
 *      alert('[from ready][open-book-make-comment][then] ' + JSON.stringify(info, null, 2));
 *    }).catch(function (err) {
 *      alert('[from ready][open-book-make-comment][error] ' + JSON.stringify(err, null, 2));
 *    });
 * }, false);
 *
 * @todo android端测试未通过：调用失败（ps：aya项目中显示android端调用openCommentPage）
 */
export const openBookMakeComment = generateAPI({
  wv: {
    ios: {
      className: WV_CLS_IOS_NAVIGATOR,
      methodName: 'openAppBookMakeComment',
      preParam,
    },
  },
  ios: {
    preParam,
    bridgeName: 'openAppBookMakeComment',
  },
});
